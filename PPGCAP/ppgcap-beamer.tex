% Página de título
\defbeamertemplate*{title page}{texsx}[1][]{ 

	% Figuras de logo
	\begin{beamercolorbox}{logo}		
		\centering
		\begin{columns}
			\begin{column}{0.5\textwidth}
				\includegraphics[scale=0.1]{fig/unipampa.png}
			\end{column}
			\begin{column}{0.5\textwidth}  %%<--- here
				\begin{flushright}
					\includegraphics[scale=0.1]{fig/embrapa.png}
				\end{flushright}
			\end{column}
		\end{columns}
	\end{beamercolorbox}%
	
	\vskip1cm%
	
	% Título do slide
	\begin{beamercolorbox}{title page header}	
		\centering
		\usebeamerfont{title}\inserttitle\par%
	\end{beamercolorbox}%
	
	\vskip0.5cm%
	
	% Nome do curso
	\begin{beamercolorbox}{title page header}	
		\centering
		\insertinstitute\par%
	\end{beamercolorbox}%

	\vskip0.5cm%
	
	% Nome do autor
	\begin{beamercolorbox}{title page header}	
		\centering
		\insertauthor\par%
	\end{beamercolorbox}%

	\vskip1cm%
	
	% Data
	\begin{beamercolorbox}{title page header}	
		\centering
		\insertdate\par%
	\end{beamercolorbox}%
	
	\vfill
}

	\end{beamercolorbox}%
}
